function doPost(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let result = { status: 'success' };
  
  try {
    const payload = JSON.parse(e.postData.contents);
    const action = payload.action;

    if (action === 'getData') {
      result.data = getAllData(ss);
    } else if (action === 'addData') {
      
      let fileUrl = "";
      if (payload.fileData) {
         fileUrl = uploadToDrive(payload.fileData);
         payload.data.ktpUrl = fileUrl; // Otomatis disisipkan ke data sebelum di-save!
      }
      
      addDataDynamically(ss, payload.sheetName, payload.data);
      result.message = "Data berhasil ditambahkan";
      result.url = fileUrl;
      
    } else if (action === 'updateData') {
      
      let fileUrl = payload.data.ktpUrl || "";
      if (payload.fileData) {
         fileUrl = uploadToDrive(payload.fileData);
         payload.data.ktpUrl = fileUrl; 
      }
      
      updateDataDynamically(ss, payload.sheetName, payload.data);
      result.message = "Data otomatis diperbarui";
      result.url = fileUrl;
      
    } else if (action === 'deleteData') {
      deleteDataDynamically(ss, payload.sheetName, payload.data);
      result.message = "Data berhasil dihapus dari database";
    }
  } catch (error) {
    result = { status: 'error', message: error.toString() };
  }

  return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON);
}

function uploadToDrive(fileData) {
  let folder;
  const folderIter = DriveApp.getFoldersByName("ERP_KTP_Uploads");
  if (folderIter.hasNext()) {
    folder = folderIter.next();
  } else {
    folder = DriveApp.createFolder("ERP_KTP_Uploads");
    folder.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
  }
  
  // Hapus file lama jika ada (agar Google Drive tidak penuh)
  if (fileData.oldUrl) {
    try {
      const match = fileData.oldUrl.match(/id=([a-zA-Z0-9_-]+)/);
      if (match && match[1]) {
        DriveApp.getFileById(match[1]).setTrashed(true);
      }
    } catch(e) {
      // Abaikan jika file lama tidak dapat ditemukan
    }
  }

  const blob = Utilities.newBlob(Utilities.base64Decode(fileData.base64), fileData.mimeType, fileData.filename);
  const file = folder.createFile(blob);
  return "https://drive.google.com/uc?export=view&id=" + file.getId();
}

function setupDatabase(ss) {
  const tables = {
    'Cabang': ['id', 'namaCabang', 'lokasi', 'tipeCuti'],
    'Owner': ['id', 'nama', 'posisi', 'username', 'password'],
    'Karyawan': ['id', 'nama', 'nik', 'posisi', 'status', 'gajiPokok', 'tunjangan', 'username', 'password', 'role', 'cabangId', 'noTelp', 'tanggalGabung', 'noRekening', 'jatahCuti', 'ktpUrl'],
    'Keuangan': ['id', 'tanggal', 'tipe', 'keterangan', 'nominal', 'statusTransaksi', 'cabangId'],
    'Laporan': ['id', 'waktu', 'judul', 'detail', 'cabangId'],
    'RiwayatPosisi': ['id', 'karyawanId', 'tanggal', 'posisiLama', 'posisiBaru', 'keterangan'],
    'Sistem': ['id', 'kunci', 'nilai']
  };

  for (let sheetName in tables) {
    let sheet = ss.getSheetByName(sheetName);
    let isNewSheet = false;
    
    if (!sheet) {
      sheet = ss.insertSheet(sheetName);
      isNewSheet = true;
    }
    
    const requiredHeaders = tables[sheetName];
    let currentHeaders = [];
    const lastCol = sheet.getLastColumn();
    
    if (lastCol > 0) {
       currentHeaders = sheet.getRange(1, 1, 1, lastCol).getValues()[0];
    }

    if (currentHeaders.length === 0 || currentHeaders[0] === "") {
      sheet.getRange(1, 1, 1, requiredHeaders.length).setValues([requiredHeaders]);
      sheet.getRange(1, 1, 1, requiredHeaders.length).setFontWeight("bold");
      
      if (isNewSheet) {
        if (sheetName === 'Sistem') sheet.appendRow(['SYS-1', 'maintenance_mode', 'false']);
        if (sheetName === 'Owner') sheet.appendRow(['OWN-1', 'Super Admin', 'Owner', 'admin', 'admin123']);
      }
    } else {
      let newColIndex = lastCol + 1;
      requiredHeaders.forEach(header => {
        if (currentHeaders.indexOf(header) === -1) {
          sheet.getRange(1, newColIndex).setValue(header);
          newColIndex++;
        }
      });
    }
  }
}

function getAllData(ss) {
  setupDatabase(ss); 
  let allData = {};
  
  const sheets = ss.getSheets();
  sheets.forEach(sheet => {
    const sheetName = sheet.getName();
    const lastRow = sheet.getLastRow();
    const lastCol = sheet.getLastColumn();
    
    if (lastRow > 1 && lastCol > 0) {
      const data = sheet.getRange(1, 1, lastRow, lastCol).getValues();
      const headers = data[0]; 
      let rows = [];
      
      for (let i = 1; i < data.length; i++) {
        let rowObj = {};
        let isEmptyRow = true; 
        
        headers.forEach((header, index) => {
          if (header !== "") {
            const val = data[i][index];
            rowObj[header] = val;
            if (val !== "" && val !== null && val !== undefined) {
               isEmptyRow = false;
            }
          }
        });
        
        if (!isEmptyRow) {
          rows.push(rowObj);
        }
      }
      allData[sheetName.toLowerCase()] = rows;
    } else {
      allData[sheetName.toLowerCase()] = [];
    }
  });
  return allData;
}

function addDataDynamically(ss, sheetName, newRowData) {
  setupDatabase(ss); 
  let sheet = ss.getSheetByName(sheetName);
  if(!sheet) throw new Error("Sheet tidak ditemukan!");
  
  const lastCol = sheet.getLastColumn();
  const headers = sheet.getRange(1, 1, 1, lastCol).getValues()[0];
  let newRow = new Array(headers.length).fill("");
  
  for (let key in newRowData) {
    let colIndex = headers.indexOf(key);
    if (colIndex !== -1) {
      newRow[colIndex] = newRowData[key];
    } else {
      // Keamanan jika kolom entah bagaimana belum terbaca
      sheet.getRange(1, headers.length + 1).setValue(key);
      headers.push(key);
      newRow.push(newRowData[key]);
    }
  }
  
  sheet.appendRow(newRow);
}

function updateDataDynamically(ss, sheetName, rowData) {
  setupDatabase(ss); 
  let sheet = ss.getSheetByName(sheetName);
  if(!sheet) throw new Error("Sheet tidak ditemukan!");
  
  const lastRow = sheet.getLastRow();
  const lastCol = sheet.getLastColumn();
  
  const data = sheet.getRange(1, 1, lastRow, lastCol).getValues();
  const headers = data[0];
  const idColIndex = headers.indexOf('id');
  
  if (idColIndex === -1) throw new Error("Kolom id tidak ditemukan untuk pencocokan data");
  
  for (let i = 1; i < data.length; i++) {
    if (data[i][idColIndex] === rowData.id) {
      let newRow = [...data[i]]; 
      for (let key in rowData) {
        let colIndex = headers.indexOf(key);
        if (colIndex !== -1) {
          newRow[colIndex] = rowData[key];
        }
      }
      sheet.getRange(i + 1, 1, 1, newRow.length).setValues([newRow]);
      return;
    }
  }
  throw new Error("ID data tidak ditemukan, gagal update.");
}

function deleteDataDynamically(ss, sheetName, id) {
  setupDatabase(ss);
  let sheet = ss.getSheetByName(sheetName);
  if(!sheet) throw new Error("Sheet tidak ditemukan!");
  
  const lastRow = sheet.getLastRow();
  const lastCol = sheet.getLastColumn();
  const data = sheet.getRange(1, 1, lastRow, lastCol).getValues();
  const headers = data[0];
  const idColIndex = headers.indexOf('id');
  
  if (idColIndex === -1) throw new Error("Kolom id tidak ditemukan!");
  
  for (let i = 1; i < data.length; i++) {
    if (data[i][idColIndex] === id) {
      sheet.deleteRow(i + 1);
      return;
    }
  }
  throw new Error("ID data tidak ditemukan, gagal dihapus.");
}
